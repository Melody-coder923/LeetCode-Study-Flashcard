Top-down DP  递归 + 记忆化
```
from functools import lru_cache

class Solution:
    def minDistance(self, word1: str, word2: str) -> int:

        @lru_cache(None)
        def dfs(i: int, j: int) -> int:
            # 如果有一个字符串为空，返回另一个字符串的长度
            if i == len(word1):
                return len(word2) - j
            if j == len(word2):
                return len(word1) - i

            if word1[i] == word2[j]:
                return dfs(i + 1, j + 1)
            else:
                insert = dfs(i, j + 1)
                delete = dfs(i + 1, j)
                replace = dfs(i + 1, j + 1)
                return 1 + min(insert, delete, replace)

        return dfs(0, 0)
```

Bottom-up DP 动态规划
```
class Solution:
    def minDistance(self, word1: str, word2: str) -> int:
        m, n = len(word1), len(word2)
        dp = [[0] * (n + 1) for _ in range(m + 1)]

        # 初始化边界：空字符串转化为另一个
        for i in range(m + 1):
            dp[i][0] = i
        for j in range(n + 1):
            dp[0][j] = j

        # 填表
        for i in range(1, m + 1):
            for j in range(1, n + 1):
                if word1[i - 1] == word2[j - 1]:
                    dp[i][j] = dp[i - 1][j - 1]
                else:
                    dp[i][j] = 1 + min(
                        dp[i - 1][j],    # 删除
                        dp[i][j - 1],    # 插入
                        dp[i - 1][j - 1] # 替换
                    )

        return dp[m][n]
```
